      def __str__(self):
      	  str = "Image:" + self.name + " - " + ':'.join(self.dim)
	  if self.tn:
	     str = str + " thumb: " + self.tn
	  else:
	     str = str + " thumb: none"
	  return str
      def thumb_src(self):
          str = '<img src="' + self.tn + '">'
	  return str
      def html_src(self):
          str = '<img src="' + self.name + '">'
	  if self.tn:
	     str = '<a href="' + self.name + '">' + self.thumb_src() + '</a>'
   	  return str
